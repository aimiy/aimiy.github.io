## 下拉框懒加载

```vue
<template>
  <el-select
    v-model="_value"
    :placeholder="placeholder"
    filterable
    remote
    :remote-method="remoteGet"
    clearable
    v-el-select-loadmore="loadMore"
    @clear="initForm"
    :multiple="multiple"
    :collapse-tags="collapseTags"
    :loading="loading"
  >
    <slot />
  </el-select>
</template>

<script>
export default {
  directives: {
    "el-select-loadmore": {
      bind(el, binding) {
        const SELECTWRAP_DOM = el.querySelector(
          ".el-select-dropdown .el-select-dropdown__wrap"
        );
        SELECTWRAP_DOM.addEventListener("scroll", function() {
          const condition =
            this.scrollHeight - this.scrollTop <= this.clientHeight;
          if (condition) {
            binding.value();
          }
        });
      }
    }
  },
  props: {
    value: { default: null },
    getList: { type: Function },
    placeholder: { type: String, default: "请选择" },
    start: {
      type: Number,
      default: 1
    },
    limit: {
      type: Number,
      default: 20
    },
    totalPages: {
      type: Number,
      default: 0
    },
    options: {
      type: Array,
      default: []
    },
    multiple: {
      type: Boolean,
      default: false
    },
    collapseTags: {
      type: Boolean,
      default: false
    },
    loading: {
      type: Boolean,
      default: false
    }
  },
  data() {
    return {
      searchinfo: "",
      startValue: this.start
    };
  },
  computed: {
    _value: {
      get() {
        return this.value;
      },
      set(val) {
        this.$emit("input", val);
      }
    }
  },
  methods: {
    initForm() {
      this._value = null;
      this.remoteGet();
    },
    loadMore() {
      if (this.startValue < this.totalPages) {
        this.$emit("update:start", ++this.startValue);
        this.$emit("getList", this.searchinfo);
      }
    },
    remoteGet(searchinfo) {
      this.startValue = 1;
      this.$emit("update:start", 1);
      this.$emit("update:options", []);
      this.searchinfo = searchinfo;
      this.$emit("getList", this.searchinfo);
    }
  }
};
</script>
```

## 水印

```vue
<template>
<div></div>
</template>
<script>
export default {
  props: {
    // // // 显示的水印文本
    // inputText: {
    //   type: String,
    //   default: ''
    // },
    // 是否允许通过js或开发者工具等途径修改水印DOM节点（水印的id，attribute属性，节点的删除）
    // true为允许，默认不允许
    inputAllowDele: {
      type: Boolean,
      default: false
    },
    // 是否在组件销毁时去除水印节点（前提是允许用户修改DOM，否则去除后会再次自动生成）
    // true会，默认不会
    inputDestroy: {
      type: Boolean,
      default: false
    }
  },
  data () {
      var date = new Date();
      return {
        inputText:(window.localStorage.company || '') + '  ' + window.localStorage.userName + '  ' + date.getFullYear() + '-' + (date.getMonth() + 1)+'-'+date.getDate(),
      maskDiv: {} // 当前显示的水印div节点DOM对象
    }
  },
  mounted () {
    // 确认DOM渲染后再执行
    this.$nextTick(() => {
      // 创建水印节点
      this.init()
      if (!this.inputAllowDele) {
        // 设置水印节点修改的DOM事件
        this.Monitor()
      }
    })
  },
  methods: {
    init () {
      let canvas = document.createElement('canvas')
      canvas.id = 'canvas'
      canvas.width = '450' // 单个水印的宽高
      canvas.height = '180'
      this.maskDiv = document.createElement('div')
      let ctx = canvas.getContext('2d')
      ctx.font = 'normal 18px Microsoft Yahei' // 设置样式
      ctx.fillStyle = 'rgba(112, 113, 114, 0.2)' // 水印字体颜色
      ctx.fillText(this.inputText, 30, 20)
      let src = canvas.toDataURL('image/png')
      this.maskDiv.style.position = 'fixed'
      this.maskDiv.style.zIndex = '9999'
      this.maskDiv.id = '_waterMark'
      this.maskDiv.style.top = '30px'
      this.maskDiv.style.left = '0'
      this.maskDiv.style.height = '100%'
      this.maskDiv.style.width = '100%'
      this.maskDiv.style.webkitTransform = "rotate(-20deg)";
      this.maskDiv.style.MozTransform = "rotate(-20deg)";
      this.maskDiv.style.msTransform = "rotate(-20deg)";
      this.maskDiv.style.OTransform = "rotate(-20deg)";
      this.maskDiv.style.transform = "rotate(-20deg)";
      this.maskDiv.style.pointerEvents = 'none'
      this.maskDiv.style.backgroundImage = 'URL(' + src + ')'
      // 水印节点插到body下
      document.body.appendChild(this.maskDiv)
    },
    Monitor () {
      let body = document.getElementsByTagName('body')[0]
      let options = {
        childList: true,
        attributes: true,
        characterData: true,
        subtree: true,
        attributeOldValue: true,
        characterDataOldValue: true
      }
      let observer = new MutationObserver(this.callback)
      observer.observe(body, options) // 监听body节点
    },
    // DOM改变执行callback
    callback (mutations, observer) {
      // 当attribute属性被修改
      if (mutations[0].target.id === '_waterMark') {
        this.removeMaskDiv()
      }
      // 当id被改变时
      if (mutations[0].attributeName === 'id') {
        this.removeMaskDiv()
        this.init()
      }
      // 当节点被删除
      if (mutations[0].removedNodes[0] && mutations[0].removedNodes[0].id === '_waterMark') {
        this.init()
      }
    },
    /* public */
    // 手动销毁水印DOM
    removeMaskDiv () {
      document.body.removeChild(this.maskDiv)
    },
    // 手动生成水印
    createMaskDiv () {
      this.init()
    }
  },
  watch: {
    // 监听传入水印文本变化
    inputText () {
      this.$nextTick(() => {
        this.removeMaskDiv()
      })
    }
  },
  beforeDestroy () {
    // 组件销毁时去除生成在body节点下的水印节点
    if (this.inputDestroy) {
      this.removeMaskDiv()
    }
  }
}
</script>
```

```vue
<template>
  <div>
    <el-upload
      drag
      action="#"
      :limit="1"
      list-type="picture-card"
      :file-list="fileList"
      ref="imgUpload"

    </el-dialog>

export default {
  props: {
    value: {
      default: null,
    },
    accept: {
      type: String,
      default: "",
    },
    isBase64: {
      type: Boolean,
      default: false,
    },
    limtSize: {
      type: Boolean,
      default: false,
    },
  },
  data() {
    return {
      dialogImageUrl: "",
      dialogVisible: false,
      disabled: false,
      file: {},
    };
  },
  computed: {
    _value: {
      get() {
        return this.value;
      },
      set(val) {
        this.$emit("input", val);
      },
    },
    fileList() {
      if (this._value) {
        return [{ url: this._value }];
      } else {
        return [];
      }
    },
  },
  methods: {
    handleChange(file) {},
    uploadFile() {
      let file = this.file;
      if (this.isBase64) {
        let reader = new FileReader();
        reader.onload = (event) => {
          this._value = event.target.result;
        };
        reader.readAsDataURL(file);
      } else {
        this._value = file;
      }
    },
    checkSuffix(file) {
      let fileTypeArr = file.type.split("/");
      if (fileTypeArr[0] !== "image") {
        this.$message.error("上传的不是图片！");
        return false;
      }
      let acceptArr = this.accept.split(/,|\s/);
      let nameArr = file.name.split(".");
      let current = nameArr[nameArr.length - 1];
      let isMatch =
        acceptArr.findIndex((item) => {
          return item === "." + current;
        }) != -1;
      const isLt2M = file.size / 1024 <= 64;
      if (!isMatch) {
        this.$message.error("图片格式不匹配");
        return false;
      }
      if(this.limtSize) {
        if (!isLt2M) {
          this.$message.error("上传头像图片大小不能超过 64KB!");
          return false;
        }
      }
      this.file = file;
      return true;
    },
    handleRemove(file, fileList) {
      this.$refs["imgUpload"].clearFiles();
      this._value = null;
    },
    handlePictureCardPreview(file) {
      this.dialogImageUrl = file.url;
      this.dialogVisible = true;
    },

    onExceed() {
      this.$message.warning("请删除后再上传！");
    },
  },
};
```

### 综合+drag

```vue
<template>
  <div>
    <el-upload
      action="#"
      :multiple="multiple"
      :limit="limit"
      :file-list="fileList"
      ref="fileUpload"
      :accept="accept"
      :on-change="handleChange"
      :on-exceed="onExceed"
      :before-upload="checkSuffix"
      :http-request="uploadFile"
      :on-remove="handleRemove"
      :drag="drag"
    >
      <div v-if="drag">
        <i class="el-icon-upload"></i>
        <div class="el-upload__text">
          将文件拖到此处，或<em>点击上传</em
          ><span v-if="maxSize">，文件小于{{ maxSize }}M</span>
        </div>
        <div class="el-upload__tip" slot="tip">
          <p v-if="accept">格式要求：{{ accept }}</p>
        </div>
      </div>
      <div v-else>
        <el-button
          icon="el-icon-upload2"
          slot="trigger"
          size="small"
          type="primary"
          >点击上传</el-button
        >
        <div class="el-upload__tip" slot="tip">
          <p>
            <span v-if="accept">格式要求：{{ accept }}；</span
            ><span v-if="maxSize">文件小于{{ maxSize }}M</span>
          </p>
        </div>
      </div>
    </el-upload>
  </div>
</template>

<script>
export default {
  props: {
    value: {
      default: null,
    },
    accept: {
      type: String,
      default: "",
    },
    maxSize: {
      type: Number,
      default: 0,
    },
    multiple: {
      type: Boolean,
      default: false,
    },
    limit: {
      type: Number,
      default: 1,
    },
    drag: {
      type: Boolean,
      default: false,
    },
  },
  data() {
    return {
      disabled: false,
      file: {},
      fileList: [],
    };
  },
  computed: {
    _value: {
      get() {
        return this.value;
      },
      set(val) {
        this.$emit("input", val);
      },
    },
  },
  watch: {
    value(val) {
      if (!val || val.length < 1) {
        this.fileList = [];
      }
    },
  },
  methods: {
    handleChange(file) {},
    uploadFile() {
      let file = this.file;
      if (this.multiple) {
        if (!this._value) {
          this._value = [];
        }
        this._value.push(file);
        this._value = this._value;
      } else {
        this._value = file;
      }
    },
    checkSuffix(file) {
      if (this.accept) {
        let acceptArr = this.accept.split(/,|\s/);
        let nameArr = file.name.split(".");
        let current = nameArr[nameArr.length - 1];
        let isMatch =
          acceptArr.findIndex((item) => {
            return item === "." + current;
          }) != -1;
        if (!isMatch) {
          this.$message.error("文件格式不匹配");
          return false;
        }
      }
      if (this.maxSize) {
        const isLt2M = file.size / 1024 / 1024 < this.maxSize;
        if (!isLt2M) {
          this.$message.error(
            "上传头像图片大小不能超过 " + this.maxSize + "MB!"
          );
          return false;
        }
      }
      this.file = file;
      return true;
    },
    handleRemove(file, fileList) {
      if (this.multiple) {
        let index = this._value.findIndex((item) => item.name === file.name);
        if (index > -1) {
          this._value.splice(index, 1);
        }
      } else {
        this._value = null;
        this.$refs["fileUpload"].clearFiles();
      }
    },
    onExceed() {
      this.$message.warning("请删除后再上传！");
    },
  },
};
</script>
```

```vue
      type: Array,
      default: []
    },

      type:Boolean,
      default:false
    },

      type:Function,
      default:() => {return false;}
    }
  },

      this.$router.go(0);
    }
  }
};
</script>
```
